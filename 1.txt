<script>
    function getTime(){
        var myDate = new Date();
        var myYear = myDate.getFullYear();
        var myMonth = myDate.getMonth()+1;
        var myToday = myDate.getDate();
        var myDay = myDate.getDay();
        var myHour = myDate.getHours();
        var myMinute = myDate.getMinutes();
        var mySecond = myDate.getSeconds();
        var week = ['星期日','星期一','星期二','星期三','星期四','星期五','星期六'];
        var nowTime;
        nowTime = myYear+'-'+fillZero(myMonth)+'-'+fillZero(myToday)+' '+fillZero(myHour)+':'+fillZero(myMinute)+':'+fillZero(mySecond)+' '+week[myDay];
        $('#time').html(nowTime);
    };
    function fillZero(str){
        return (str<10)?('0'+str):str;
    }
    setInterval(getTime,1000);
</script>

<script>
    const items = document.querySelectorAll('.slider-list li');
    const panels = document.querySelectorAll('.tab-panel');
    const prev = document.querySelector('.slider-btn.prev');
    const next = document.querySelector('.slider-btn.next');
    let current = 1;

    function updateSlider(index) {
        items.forEach((item, i) => {
            item.classList.toggle('active', i === index);
            panels[i].style.display = i === index ? 'block' : 'none';
        });
    }

    items.forEach((item, index) => {
        item.addEventListener('click', () => {
            current = index;
            updateSlider(current);
        });
    });

    prev.addEventListener('click', () => {
        current = (current - 1 + items.length) % items.length;
        updateSlider(current);
    });

    next.addEventListener('click', () => {
        current = (current + 1) % items.length;
        updateSlider(current);
    });

    // 初始化显示第二个页面（修复数据）
    panels.forEach((panel, i) => {
        panel.style.display = i === 1 ? 'block' : 'none';
    });
    updateSlider(current);
</script>
